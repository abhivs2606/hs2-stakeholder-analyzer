    "## Step 4 — Extract text (with OCR fallback)\n",
    "First tries direct text extraction. If the PDF has image-based pages, falls back to OCR automatically."
   ]
  },
  {
   "cell_type": "code",
   "execution_count": null,
   "metadata": {},
   "outputs": [],
   "source": [
    "import fitz\n",
    "import pytesseract\n",
    "from pdf2image import convert_from_path\n",
    "\n",
    "def extract_text_with_ocr_fallback(filepath):\n",
    "    \"\"\"\n",
    "    Try direct extraction first.\n",
    "    If a page returns less than 50 chars, treat it as image-based and OCR it.\n",
    "    \"\"\"\n",
    "    doc = fitz.open(filepath)\n",
    "    full_text = ''\n",
    "    needs_ocr_pages = []\n",
    "\n",
    "    for i, page in enumerate(doc):\n",
    "        text = page.get_text().strip()\n",
    "        if len(text) > 50:\n",
    "            full_text += text + ' '\n",
    "        else:\n",
    "            needs_ocr_pages.append(i)\n",
    "\n",
    "    if needs_ocr_pages:\n",
    "        print(f'  Pages needing OCR: {needs_ocr_pages}')\n",
    "        images = convert_from_path(filepath, dpi=200)\n",
    "        for i in needs_ocr_pages:\n",
    "            if i < len(images):\n",
    "                ocr_text = pytesseract.image_to_string(images[i])\n",
    "                full_text += ocr_text + ' '\n",
    "                print(f'  OCR page {i+1}: extracted {len(ocr_text)} chars')\n",
    "\n",
    "    return full_text.strip()\n",
    "\n",
    "\n",
    "def chunk_text(text, chunk_size=600):\n",
    "    words = text.split()\n",
    "    chunks = []\n",
    "    current = []\n",
    "    length = 0\n",
    "    for word in words:\n",
    "        current.append(word)\n",
    "        length += len(word) + 1\n",
    "        if length >= chunk_size:\n",
    "            chunks.append(' '.join(current))\n",
    "            current = []\n",
    "            length = 0\n",
    "    if current:\n",
    "        chunks.append(' '.join(current))\n",
    "    return chunks\n",
    "\n",
    "\n",
    "print('Extracting text from PDF...')\n",
    "full_text = extract_text_with_ocr_fallback(pdf_filename)\n",
    "chunks = chunk_text(full_text)\n",
    "\n",
    "print(f'\\n✅ Total text extracted: {len(full_text)} characters')\n",
    "print(f'✅ Split into {len(chunks)} chunks')\n",
    "print(f'\\nPreview:\\n{full_text[:400]}')"
   ]
  },

    "## Step 5 — Define detection functions"
   ]
  },
  {
   "cell_type": "code",
   "execution_count": null,
   "metadata": {},
   "outputs": [],
   "source": [
    "STAKEHOLDER_LABELS = [\n",
    "    'Government and policy makers',\n",
    "    'Local residents and communities',\n",
    "    'Environmental groups',\n",
    "    'Business and industry',\n",
    "    'Transport and infrastructure experts',\n",
    "    'Local councils and authorities'\n",
    "]\n",
    "\n",
    "PRIORITY_LABELS = [\n",
    "    'Cost and budget concerns',\n",
    "    'Environmental impact',\n",
    "    'Community disruption',\n",
    "    'Project timeline and delivery',\n",
    "    'Economic growth and connectivity',\n",
    "    'Property and land rights'\n",
    "]\n",
    "\n",
    "\n",
    "def detect_stakeholder(chunk):\n",
    "    result = classifier(chunk, STAKEHOLDER_LABELS, multi_label=False)\n",
    "    return result['labels'][0], round(result['scores'][0], 2)\n",
    "\n",
    "\n",
    "def detect_priority(chunk):\n",
    "    result = classifier(chunk, PRIORITY_LABELS, multi_label=False)\n",
    "    return result['labels'][0], round(result['scores'][0], 2)\n",
    "\n",
    "\n",
    "def detect_stance(chunk):\n",
    "    result = sentiment(chunk[:512])\n",
    "    label = result[0]['label']\n",
    "    score = result[0]['score']\n",
    "    if score < 0.65:\n",
    "        return 'Mixed'\n",
    "    return 'Support' if label == 'POSITIVE' else 'Oppose'\n",
    "\n",
    "\n",
    "print('✅ Detection functions ready')"
   ]
  },

    "## Step 6 — Run analysis across all chunks"
   ]
  },
  {
   "cell_type": "code",
   "execution_count": null,
   "metadata": {},
   "outputs": [],
   "source": [
    "all_results = []\n",
    "\n",
    "print(f'Analyzing {len(chunks)} chunks...\\n')\n",
    "\n",
    "for i, chunk in enumerate(chunks):\n",
    "    print(f'  Chunk {i+1}/{len(chunks)}...', end=' ')\n",
    "\n",
    "    stakeholder, s_score = detect_stakeholder(chunk)\n",
    "    priority, p_score = detect_priority(chunk)\n",
    "    stance = detect_stance(chunk)\n",
    "\n",
    "    result = {\n",
    "        'chunk_index': i,\n",
    "        'stakeholder_group': stakeholder,\n",
    "        'stakeholder_confidence': s_score,\n",
    "        'priority_dimension': priority,\n",
    "        'priority_confidence': p_score,\n",
    "        'position': stance,\n",
    "        'text_preview': chunk[:150]\n",
    "    }\n",
    "\n",
    "    all_results.append(result)\n",
    "    print(f'[{stakeholder[:25]}] | [{priority[:25]}] | {stance}')\n",
    "\n",
    "print(f'\\n✅ Done. {len(all_results)} chunks analyzed.')"
   ]

    "## Step 7 — Print the conflict report"
   ]
  },
  {
   "cell_type": "code",
   "execution_count": null,
   "metadata": {},
   "outputs": [],
   "source": [
    "from collections import defaultdict\n",
    "\n",
    "def build_and_print_report(results):\n",
    "    filtered = [r for r in results if r['stakeholder_confidence'] >= 0.4]\n",
    "\n",
    "    by_stakeholder = defaultdict(list)\n",
    "    by_priority = defaultdict(list)\n",
    "\n",
    "    for r in filtered:\n",
    "        by_stakeholder[r['stakeholder_group']].append(r)\n",
    "        by_priority[r['priority_dimension']].append(r)\n",
    "\n",
    "    print('=' * 65)\n",
    "    print('HS2 STAKEHOLDER CONFLICT REPORT')\n",
    "    print('=' * 65)\n",
    "\n",
    "    print('\\n📋 STAKEHOLDER POSITIONS\\n')\n",
    "    for group, items in by_stakeholder.items():\n",
    "        stances = [r['position'] for r in items]\n",
    "        dominant = max(set(stances), key=stances.count)\n",
    "        support = stances.count('Support')\n",
    "        oppose = stances.count('Oppose')\n",
    "        mixed = stances.count('Mixed')\n",
    "        print(f'  {group}')\n",
    "        print(f'  → Dominant stance: {dominant}  |  ✅ {support}  ❌ {oppose}  〰️ {mixed} chunks')\n",
    "        print()\n",
    "\n",
    "    print('\\n🔢 PRIORITY DIMENSION BREAKDOWN\\n')\n",
    "    for priority, items in by_priority.items():\n",
    "        stances = [r['position'] for r in items]\n",
    "        support = stances.count('Support')\n",
    "        oppose = stances.count('Oppose')\n",
    "        mixed = stances.count('Mixed')\n",
    "        print(f'  {priority}')\n",
    "        print(f'  → ✅ {support} support  |  ❌ {oppose} oppose  |  〰️ {mixed} mixed')\n",
    "        print()\n",
    "\n",
    "    print('\\n⚡ CONFLICT ZONES\\n')\n",
    "    found = False\n",
    "    for priority, items in by_priority.items():\n",
    "        stances = set(r['position'] for r in items)\n",
    "        if 'Support' in stances and 'Oppose' in stances:\n",
    "            found = True\n",
    "            print(f'  ⚡ {priority}')\n",
    "            for r in items:\n",
    "                icon = '✅' if r['position'] == 'Support' else '❌' if r['position'] == 'Oppose' else '〰️'\n",
    "                print(f'     {icon} [{r[\"stakeholder_group\"][:30]}]: \"{r[\"text_preview\"][:80]}...\"')\n",
    "            print()\n",
    "\n",
    "    if not found:\n",
    "        print('  No direct conflicts detected.')\n",
    "        print('  This document is a government summary — not raw consultation responses.')\n",
    "        print('  For richer conflicts, use a multi-respondent consultation corpus.')\n",
    "\n",
    "    print('\\n' + '=' * 65)\n",
    "\n",
    "\n",
    "build_and_print_report(all_results)"
   ]
